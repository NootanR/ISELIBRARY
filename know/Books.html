 31     public ArrayList<booktable> getbooktablelist(){
 32         ArrayList<booktable> booktablelist = new ArrayList<booktable>();
 33         Connection conn;
 34         conn = Connector.obtain();
 35         String query = "select * from `book` where availabale='yes'";
 36         Statement st;
 37         ResultSet rs;
 38         try
 39         {
 40         st= conn.createStatement();
 41         rs=st.executeQuery(query);
 42         booktable b;
 43         while(rs.next()){
 44             b =new booktable(rs.getInt("isbn"),rs.getString("bname"),rs.getString("pubname"),rs.getString("autname"),rs.getInt("price")); 
 45             booktablelist.add(b);
 46         }
 47         
 48         }
 49         catch(Exception e){
 50         }
 51         return booktablelist;        
 52     }

 54     public void showinjtable(){
 55         ArrayList<booktable> list = getbooktablelist();
 56         DefaultTableModel model;
 57         model = (DefaultTableModel)jTable1_display.getModel();
 58         Object[] row= new Object[5];
 59         for(int i=0;i<list.size();i++){
 60             row[0]= list.get(i).getID();
 61             row[1]= list.get(i).getBookname();
 62             row[2]= list.get(i).getPubname();
 63             row[3]= list.get(i).getAuthname();
 64             row[4]= list.get(i).getprice();
 65             model.addRow(row);
 66         }
 67     
 68     }
